Class Phalcon\Debug\Dump

Dumps information about a variable(s)

<?php

$foo = 123;
echo (new \Phalcon\Debug\Dump())->var($foo, "foo");

<?php

$foo = "string";
$bar = ["key" => "value"];
$baz = new stdClass();
echo (new \Phalcon\Debug\Dump())->vars($foo, $bar, $baz);
